EKCalendarChooser : UIViewController, EKCalendarChooserDelegate

    // Called whenever the selection is changed by the user
    @available(iOS 5.0, *)
    optional public func calendarChooserSelectionDidChange(_ calendarChooser: EKCalendarChooser)

    
    // These are called when the corresponding button is pressed to dismiss the

    @available(iOS 5.0, *)

    @available(iOS 5.0, *)
    optional public func calendarChooserDidCancel(_ calendarChooser: EKCalendarChooser)
